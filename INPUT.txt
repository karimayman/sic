COPY START 0
FIRST STX RETADR
+LDA LENGTH
BASE LENGTH
CLOOP +JSUB RDREC
LDA LENGTH
COMP 0
JEQ ENDFIL
+JSUB @WRREC
J CLOOP
ENDFIL LDA =C'EOF'
STA BUFFER
LDA 3
STA LENGTH
+JSUB WRREC
J RETADR
LTORG
RETADR RESW 1
LENGTH RESW 1
BUFFER RESB 4096
BUFEND EQU *
MAXLEN EQU BUFEND-BUFFER
RDREC CLEAR X
CLEAR A
CLEAR S
+LDT MAXLENGTH
RLOOP TD INPUT
JEQ RLOOP
RD INPUT
COMPR A S
JEQ EXIT
TIXR T
JLT RLOOP
EXIT STX =X'F1'
RSUB
INPUT BYTE X'F1'
WRREC CLEAR X
LDT LENGTH
WLOOP TD =X'05'
JEQ WLOOP
LTORG
LOKL TIXR T
JLT WLOOP
RSUB
END FIRST
