HW1B START 0030
WRREC +LDX ZERO
BASE WRREC
WLOOP LDX OUTPUT,X
YES JEQ @WLOOP
JEQ WLOOP
JEQ @WLOOP
JEQ #WLOOP
LDCH RECORD
WD OUTPUT
TIX LENGTH
JLT WLOOP
ZERO WORD FFFF
LENGTH WORD 1
CLEAR X X
TIO
OUTPUT BYTE C'5'
RECORD RESB 100
BUFFER RESB 4096
BUFEND EQU *
MAXLEN EQU BUFEND-BUFFER
END 0030
